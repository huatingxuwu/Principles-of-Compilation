  开始符号: S
  非终结符: ['L', 'R', 'S']
  终结符:   ['*', '=', 'id']
  产生式:
  S → L = R
  S → R
  L → R * R
  L → id
  R → id

增广文法: 添加 S' → S

初始项: S' → · S

计算 CLOSURE({S' → · S}):
      由 S' → · S 预测: 加入 S → · L = R
      由 S' → · S 预测: 加入 S → · R
      由 S → · R 预测: 加入 R → · id
      由 S → · L = R 预测: 加入 L → · R * R
      由 S → · L = R 预测: 加入 L → · id
  → I₀ = {L → · R * R,  L → · id,  R → · id,  S → · L = R,  S → · R,  S' → · S}

── 处理状态 I0 ──
  项集: {L → · R * R,  L → · id,  R → · id,  S → · L = R,  S → · R,  S' → · S}

  GOTO(I0, 'L'):
      kernel 项 (点右移过 'L'):
        S → L · = R
    → 新状态 I1 = {S → L · = R}

  GOTO(I0, 'R'):
      kernel 项 (点右移过 'R'):
        L → R · * R
        S → R ·
    → 新状态 I2 = {L → R · * R,  S → R ·}

  GOTO(I0, 'S'):
      kernel 项 (点右移过 'S'):
        S' → S ·
    → 新状态 I3 = {S' → S ·}

  GOTO(I0, 'id'):
      kernel 项 (点右移过 'id'):
        L → id ·
        R → id ·
    → 新状态 I4 = {L → id ·,  R → id ·}

── 处理状态 I1 ──
  项集: {S → L · = R}

  GOTO(I1, '='):
      kernel 项 (点右移过 '='):
        S → L = · R
      由 S → L = · R 预测: 加入 R → · id
    → 新状态 I5 = {R → · id,  S → L = · R}

── 处理状态 I2 ──
  项集: {L → R · * R,  S → R ·}

  GOTO(I2, '*'):
      kernel 项 (点右移过 '*'):
        L → R * · R
      由 L → R * · R 预测: 加入 R → · id
    → 新状态 I6 = {L → R * · R,  R → · id}

── 处理状态 I3 ──
  项集: {S' → S ·}

── 处理状态 I4 ──
  项集: {L → id ·,  R → id ·}

── 处理状态 I5 ──
  项集: {R → · id,  S → L = · R}

  GOTO(I5, 'R'):
      kernel 项 (点右移过 'R'):
        S → L = R ·
    → 新状态 I7 = {S → L = R ·}

  GOTO(I5, 'id'):
      kernel 项 (点右移过 'id'):
        R → id ·
    → 新状态 I8 = {R → id ·}

── 处理状态 I6 ──
  项集: {L → R * · R,  R → · id}

  GOTO(I6, 'R'):
      kernel 项 (点右移过 'R'):
        L → R * R ·
    → 新状态 I9 = {L → R * R ·}

  GOTO(I6, 'id'):
      kernel 项 (点右移过 'id'):
        R → id ·
    → 闭包后项集 = I8（已存在）

── 处理状态 I7 ──
  项集: {S → L = R ·}

── 处理状态 I8 ──
  项集: {R → id ·}

── 处理状态 I9 ──
